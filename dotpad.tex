% Dot paper
% Based on: http://www.reddit.com/r/LaTeX/comments/28hgui/is_there_a_simple_way_of_creating_a_grid_of_dots/
\documentclass{article}

\usepackage{geometry}
\geometry{top=1in, bottom=1in, left=1cm, right=1cm}
\usepackage{tikz}
\usepackage{color}

\begin{document}

\foreach \z in {0,1,...,3} {
    \begin{tikzpicture}
        \foreach \x in {0,1,...,27} {
            \foreach \y in {0,1,...,34} {
                \fill[color=black!20!white] (\x/1.5,\y/1.5) circle (0.5mm);
            }
        }
    \end{tikzpicture}
    \newpage
}

\end{document}
